def registry_key(key)
  key = key.to_s.tr('_', '')
  @@registry.keys.detect do |ref|
    ref = ref.downcase
    snake_parts = ref.split('::')
    until(snake_parts.empty?)
      break if snake_parts.join('') == key
      snake_parts.shift
    end
    !snake_parts.empty?
  end
end

def resource(identifier, key=nil)
  res = lookup(identifier)
  if(key && res)
    res[key.to_sym]
  else
    res
  end
end
