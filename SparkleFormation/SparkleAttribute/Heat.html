def _list_join(*args)
  options = args.detect{|i| i.is_a?(::Hash) && i[:options]} || {:options => {}}
  args.delete(options)
  unless(args.size == 1)
    args = [args]
  end
  {'list_join' => [options[:options][:delimiter] || '', *args]}
end
